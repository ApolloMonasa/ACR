#include <bits/stdc++.h>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using u128 = __uint128_t;
using namespace std;



void slv(){
    ll x, y;
    cin >> x >> y;
    ll z = x << 1;
    bool fg = false;
    while (z < y) {
        if (y % z != 0) {
            fg = true;
            break;
        }
        z += x;
    }    
    cout << (fg ? "Yes" : "No") << "\n";
}

int main(){
    std::ios::sync_with_stdio(false); std::cin.tie(nullptr);std::cout.tie(nullptr);
    int t; std::cin >> t;
    while(t--) slv();
}

#include <bits/stdc++.h>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using u128 = __uint128_t;
using namespace std;



void slv(){
    int n; cin >> n;
    string l1, l2; cin >> l1 >> l2;
    vector<int> dp(n + 1, 0);
    dp[1] = (l1[0] != l2[0]);
    dp[2] = min(dp[1] + (l1[1] != l2[1]),
     (l1[0] != l1[1]) + (l2[0] != l2[1]));
    for (int i = 2; i <= n; i++) {
        dp[i] = min(dp[i - 1] + (l1[i - 1] != l2[i - 1]),
         dp[i - 2] + (l1[i - 2] != l1[i - 1]) + (l2[i - 2] != l2[i - 1]));
    }
    cout << dp[n] << "\n";
}

int main(){
    std::ios::sync_with_stdio(false); std::cin.tie(nullptr);std::cout.tie(nullptr);
    int t; std::cin >> t;
    while(t--) slv();
}

D

  • Key idea:前缀异或和
  • Complexity: difficult
  • Notes:难在发现前缀异或和的模4规律

我们采用一个常用的技巧，定义前缀异或和：

$$ \texttt{pref}_{i} = 1 \oplus 2 \oplus \cdots \oplus i $$

这样我们可以O(1)地获取($l, r$)区间的异或和:

$$ \texttt{pref}_{l - 1} \oplus \texttt{pref}_{r} $$

因此如果要使得区间异或和为0,则只要：

$$ \texttt{pref}_{l - 1} = \texttt{pref}_{r} $$

其中：

$$ 0 \le l - 1 \lt x \le r \le n $$

并且我们注意到:

$$ \texttt{pref}_{i} = \begin{cases} i, & \quad \text{if } i \bmod 4 = 0 \\ 1, & \quad \text{if } i \bmod 4 = 1 \\ i + 1, & \quad \text{if } i \bmod 4 = 2 \\ 0, & \quad \text{if } i \bmod 4 = 3 \end{cases} $$

可以看到，只有0和1两个值不断重复，这意味着我们要的组合只能是两个0或两个1:

$$ \texttt{pref}_{l - 1} = \texttt{pref}_{r} = 0 \\ \texttt{pref}_{l - 1} = \texttt{pref}_{r} = 1 $$

#include <bits/stdc++.h>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using u128 = __uint128_t;
using namespace std;
const ll MOD = 998'244'353L;


void slv(){
    ll n, x; cin >> n >> x;
    // 获取从0到i中数字(%4余3，即prexor为0)的数的个数
    auto get0 = [&](ll i) -> ll {
        // 多加一个是因为0也满足条件
        return (1LL + (i >= 3 ? (i - 3) / 4 + 1 : 0)) % MOD;
    };
    // 获取从0到i中数字(%4余1，即prexor为1)的数的个数
    auto get1 = [&](ll i) -> ll {
        return (i >= 1 ? (i - 1) / 4 + 1 : 0)% MOD;
    };
    ll l0 = get0(x - 1 ), r0 = (get0(n) - l0 + MOD) % MOD;
    ll ans = l0 * r0 % MOD;
    ll l1 = get1(x - 1), r1 = (get1(n) - l1 + MOD) % MOD;
    ans = (ans + l1 * r1) % MOD;
    cout << ans << "\n"; 
}

int main(){
    std::ios::sync_with_stdio(false); std::cin.tie(nullptr);std::cout.tie(nullptr);
    int t; std::cin >> t;
    while(t--) slv();
}
