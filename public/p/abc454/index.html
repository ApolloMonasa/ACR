#include <bits/stdc++.h>
#include <vector>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using u128 = __uint128_t;
using namespace std;

int main(){
    std::ios::sync_with_stdio(false); std::cin.tie(nullptr);std::cout.tie(nullptr);
    int n, m;
    cin >> n >> m;
    vector<int> a(m), b(m);
    for (int i = 0; i < m; i++) {
        cin >> a[i] >> b[i];
    }
    // dfs遍历找最深的点，定义lambda函数
    //这是从a[i]->b[i]的单向图，便利所有可能路径，不是找最长路径，是找能去的最多的点数量，可能有循环
    vector<vector<int>> graph(n);
    for (int i = 0; i < m; i++) {
        graph[a[i] - 1].push_back(b[i] - 1);
    }
    vector<bool> visited(n, false);
    int max_ic = 0;
    function<void(int)> dfs = [&](int node) {
        visited[node] = true;
        max_ic++;
        for (int neighbor : graph[node]) {
            if (!visited[neighbor]) {
                dfs(neighbor);
                // visited[neighbor] = false; // 回溯，允许再次访问
            }
        }
    };
    dfs(0);
    cout << max_ic << endl;
}

#include <algorithm>
#include <chrono>
#include <functional>
#include <iostream>
#include <numeric>
#include <string>
#include <tuple>
#include <vector>

using namespace std;

// 暴力搜索时的统计信息，便于 bench 观察复杂度与超限情况。
struct BruteStats {
    long long nodes = 0;
    bool hit_node_limit = false;
    bool hit_time_limit = false;
    double elapsed_ms = 0.0;
};

// 暴力搜索的保护阈值：防止单组跑太久。
struct BruteConfig {
    long long node_limit = 2'000'000;
    int time_limit_ms = 300;
};

static inline int idx(int r, int c, int n) {
    return r * n + c;
}

// 检查二分图颜色必要条件。
// 网格按 (r+c)%2 染色，哈密顿路径的两端颜色与黑白数量差必须匹配。
// 这里只做“必要不充分”过滤，用来加速暴力搜索。
static bool endpoint_color_condition_ok(int n, int a, int b) {
    // (a,b) is 1-based, convert to 0-based for color parity.
    int br = a - 1;
    int bc = b - 1;

    long long black = 0;
    long long white = 0;
    for (int r = 0; r < n; ++r) {
        for (int c = 0; c < n; ++c) {
            if (r == br && c == bc) continue;
            if (((r + c) & 1) == 0) {
                ++black;
            } else {
                ++white;
            }
        }
    }

    int s_color = 0;  // (0,0)
    int t_color = ((n - 1) + (n - 1)) & 1;
    long long diff = black - white;

    if (diff == 0) {
        return s_color != t_color;
    }
    if (diff == 1) {
        return s_color == 0 && t_color == 0;
    }
    if (diff == -1) {
        return s_color == 1 && t_color == 1;
    }
    return false;
}

// 暴力 DFS 寻找一条路径：
// 从 (1,1) 出发，到 (N,N) 结束；
// 恰好访问除 (A,B) 之外的所有格子各一次。
// 找到则 out_path 返回方向串（UDLR）。
bool brute_search_path(int n, int a, int b, string& out_path, BruteStats& stats, const BruteConfig& cfg = {}) {
    auto time_begin = chrono::steady_clock::now();

    // Impossible if blocked cell is start or goal.
    if ((a == 1 && b == 1) || (a == n && b == n)) {
        stats.elapsed_ms = chrono::duration<double, milli>(chrono::steady_clock::now() - time_begin).count();
        return false;
    }

    if (!endpoint_color_condition_ok(n, a, b)) {
        stats.elapsed_ms = chrono::duration<double, milli>(chrono::steady_clock::now() - time_begin).count();
        return false;
    }

    int br = a - 1;
    int bc = b - 1;

    // 可访问点数是 N*N-1（扣掉障碍点），路径边数=点数-1。
    const int total_vertices = n * n - 1;
    const int target_steps = total_vertices - 1;

    vector<char> used(n * n, 0);
    used[idx(br, bc, n)] = 1;

    vector<char> path(target_steps, '?');

    const int dr[4] = {-1, 1, 0, 0};
    const int dc[4] = {0, 0, -1, 1};
    const char mv[4] = {'U', 'D', 'L', 'R'};

    auto inside = [&](int r, int c) {
        return (0 <= r && r < n && 0 <= c && c < n);
    };

    function<bool(int, int, int)> dfs = [&](int r, int c, int step) -> bool {
        ++stats.nodes;
        if (stats.nodes >= cfg.node_limit) {
            stats.hit_node_limit = true;
            return false;
        }

        auto now = chrono::steady_clock::now();
        if (chrono::duration_cast<chrono::milliseconds>(now - time_begin).count() > cfg.time_limit_ms) {
            stats.hit_time_limit = true;
            return false;
        }

        if (r == n - 1 && c == n - 1) {
            return step == target_steps;
        }

        if (step == target_steps) {
            return false;
        }

        // 剪枝1：剩余步数必须至少覆盖到终点的曼哈顿距离。
        // 剪枝2：曼哈顿距离与剩余步数奇偶性要一致（每步翻转一次奇偶）。
        int remaining = target_steps - step;
        int md = abs((n - 1) - r) + abs((n - 1) - c);
        if (md > remaining) return false;
        if (((remaining - md) & 1) != 0) return false;

        // 候选移动按 (可拓展度, 到终点距离) 排序，
        // 类似 Warnsdorff 启发：优先走“更容易被堵死”的点。
        vector<tuple<int, int, int>> cand;  // (deg, dist_to_goal, dir)
        cand.reserve(4);

        for (int k = 0; k < 4; ++k) {
            int nr = r + dr[k];
            int nc = c + dc[k];
            if (!inside(nr, nc)) continue;
            int id = idx(nr, nc, n);
            if (used[id]) continue;

            if (nr == n - 1 && nc == n - 1 && step + 1 != target_steps) {
                continue;
            }

            int deg = 0;
            for (int t = 0; t < 4; ++t) {
                int rr = nr + dr[t];
                int cc = nc + dc[t];
                if (!inside(rr, cc)) continue;
                int jd = idx(rr, cc, n);
                if (!used[jd]) ++deg;
            }

            int dist_goal = abs((n - 1) - nr) + abs((n - 1) - nc);
            cand.emplace_back(deg, dist_goal, k);
        }

        sort(cand.begin(), cand.end());

        // 依次尝试候选，标准回溯。
        for (auto [deg, dist_goal, k] : cand) {
            (void)deg;
            (void)dist_goal;
            int nr = r + dr[k];
            int nc = c + dc[k];
            int id = idx(nr, nc, n);

            used[id] = 1;
            path[step] = mv[k];

            if (dfs(nr, nc, step + 1)) return true;

            used[id] = 0;
            path[step] = '?';
        }
        return false;
    };

    used[idx(0, 0, n)] = 1;
    bool ok = dfs(0, 0, 0);

    if (ok) {
        out_path.assign(path.begin(), path.end());
    } else {
        out_path.clear();
    }

    stats.elapsed_ms = chrono::duration<double, milli>(chrono::steady_clock::now() - time_begin).count();
    return ok;
}

// 批量 bench：枚举 n=2..max_n、以及所有障碍点 (A,B)。
// O 表示该障碍位置可行，X 表示不可行。
void benchmark_pattern(int max_n, const BruteConfig& cfg = {}) {
    cout << "# benchmark maxN=" << max_n
         << " nodeLimit=" << cfg.node_limit
         << " timeLimitMs=" << cfg.time_limit_ms << '\n';

    for (int n = 2; n <= max_n; ++n) {
        int total = 0;
        int possible = 0;
        long long total_nodes = 0;
        int timeout_cases = 0;
        int node_limit_cases = 0;

        vector<string> heat(n, string(n, '.'));

        for (int a = 1; a <= n; ++a) {
            for (int b = 1; b <= n; ++b) {
                ++total;
                string path;
                BruteStats st;
                bool ok = brute_search_path(n, a, b, path, st, cfg);
                if (ok) {
                    ++possible;
                    heat[a - 1][b - 1] = 'O';
                } else {
                    heat[a - 1][b - 1] = 'X';
                }
                if (st.hit_time_limit) ++timeout_cases;
                if (st.hit_node_limit) ++node_limit_cases;
                total_nodes += st.nodes;
            }
        }

        cout << "N=" << n << " possible=" << possible << "/" << total
             << " avgNodes=" << (total == 0 ? 0 : total_nodes / total)
             << " timeoutCases=" << timeout_cases
             << " nodeLimitCases=" << node_limit_cases
             << '\n';

        // 按网格形状打印可行性热力图。
        for (int r = 0; r < n; ++r) {
            cout << "  " << heat[r] << '\n';
        }
    }
}

int main(int argc, char** argv) {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    // 命令行模式：
    // ./tapper bench [maxN] [nodeLimit] [timeLimitMs]
    // 用于探索规律，不走标准输入。
    if (argc >= 2 && string(argv[1]) == "bench") {
        int max_n = 7;
        if (argc >= 3) max_n = stoi(argv[2]);

        BruteConfig cfg;
        if (argc >= 4) cfg.node_limit = stoll(argv[3]);
        if (argc >= 5) cfg.time_limit_ms = stoi(argv[4]);

        benchmark_pattern(max_n, cfg);
        return 0;
    }

    // 标准输入模式（AtCoder 风格）：
    // T
    // N A B
    // ...
    int t;
    cin >> t;
    while (t--) {
        int n, a, b;
        cin >> n >> a >> b;

        string path;
        BruteStats st;
        BruteConfig cfg;
        cfg.node_limit = 20'000'000;
        cfg.time_limit_ms = 1500;

        bool ok = brute_search_path(n, a, b, path, st, cfg);
        if (!ok) {
            cout << "No\n";
        } else {
            cout << "Yes\n";
            cout << path << '\n';
        }
    }

    return 0;
}

#include <bits/stdc++.h>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using u128 = __uint128_t;
using namespace std;

struct Builder {
    int N;
    int A;
    int B;
    string path;
    int r = 1;
    int c = 1;

    void mv(char ch, int k = 1) {
        while (k--) {
            path.push_back(ch);
            if (ch == 'U') --r;
            else if (ch == 'D') ++r;
            else if (ch == 'L') --c;
            else ++c;
        }
    }

    // 在当前子矩形左上角开始，剥掉最上面2行：
    // 覆盖这2行后落到下一层左上角。
    void peel_top(int w) {
        mv('R', w - 1);
        mv('D');
        mv('L', w - 1);
        mv('D');
    }

    // 在当前子矩形左上角开始，剥掉最左边2列：
    // 覆盖这2列后落到下一层左上角。
    void peel_left(int h) {
        mv('D', h - 1);
        mv('R');
        mv('U', h - 1);
        mv('R');
    }

    // 核心模板1：障碍在当前子矩形 local(1,2)。
    // 子矩形大小 h*w，h,w 都是偶数。
    void core_block_12(int h, int w) {
        mv('D');
        mv('R');

        for (int col = 3; col <= w; ++col) {
            if (col & 1) {
                mv('R');
                mv('U');
            } else {
                mv('R');
                mv('D');
            }
        }

        if (h == 2) return;

        mv('D');
        for (int row = 3; row <= h; ++row) {
            if (row & 1) {
                mv('L', w - 1);
            } else {
                mv('R', w - 1);
            }
            if (row != h) mv('D');
        }
    }

    // 核心模板2：障碍在当前子矩形 local(2,1)。
    // 子矩形大小 h*w，h,w 都是偶数。
    void core_block_21(int h, int w) {
        mv('R');
        mv('D');

        for (int row = 3; row <= h; ++row) {
            if (row & 1) {
                mv('D');
                mv('L');
            } else {
                mv('D');
                mv('R');
            }
        }

        if (w == 2) return;

        mv('R');
        for (int col = 3; col <= w; ++col) {
            if (col & 1) {
                mv('U', h - 1);
            } else {
                mv('D', h - 1);
            }
            if (col != w) mv('R');
        }
    }

    bool build(string& out) {
        // 充要条件：N为偶数，且 A+B 为奇数。
        if ((N & 1) || ((A + B) % 2 == 0)) {
            return false;
        }

        int h = N;
        int w = N;
        int a = A;
        int b = B;

        // 不断剥离不含障碍的上2行。
        while (a > 2) {
            peel_top(w);
            h -= 2;
            a -= 2;
        }

        // 不断剥离不含障碍的左2列。
        while (b > 2) {
            peel_left(h);
            w -= 2;
            b -= 2;
        }

        // 此时障碍落在当前子矩形左上2x2里。
        if (a == 1 && b == 2) {
            core_block_12(h, w);
        } else if (a == 2 && b == 1) {
            core_block_21(h, w);
        } else {
            // 理论上不会到这里（因为 a+b 必为奇数）。
            return false;
        }

        out = path;
        return true;
    }
};

void slv(){
    int n, a, b;
    cin >> n >> a >> b;

    Builder bd;
    bd.N = n;
    bd.A = a;
    bd.B = b;

    string ans;
    if (!bd.build(ans)) {
        cout << "No\n";
        return;
    }

    cout << "Yes\n";
    cout << ans << "\n";
}

int main(){
    std::ios::sync_with_stdio(false); std::cin.tie(nullptr);std::cout.tie(nullptr);
    int t; std::cin >> t;
    while(t--) slv();
}
